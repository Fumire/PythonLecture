	\section{Python?}
	\begin{frame}
		\frametitle{Python?}

		\begin{figure}
			\centering
			\includegraphics[width=0.2 \linewidth]{figures/python.png}
			\caption{Logo of Python}
		\end{figure}

		Python is an interpreted, high-level programming language created by
		Guido van Rossum. It is widely used for automation, data analysis, web
		services, scientific computing, and teaching programming.
	\end{frame}

	\begin{frame}
		\frametitle{Interpreted?}

		A compiled language is usually translated into an executable file before
		you run it.

		\medskip

		An interpreted language is run by another program called an
		\textit{interpreter}. In class, the interpreter is the \texttt{python3}
		program.

		\medskip

		Python still creates bytecode internally, but beginners normally run code
		by giving a \texttt{.py} file to the interpreter.
	\end{frame}

	\begin{frame}
		\frametitle{How to Run Python?}
		There are two common ways to run Python:
		\begin{figure}
			\centering
			$\begin{array}{cc}
				\includegraphics[width=0.45 \linewidth]{figures/run1.png}
				&
				\includegraphics[width=0.45 \linewidth]{figures/run2.png}
				\\

				\mbox{(a) Python interpreter} & \mbox{(b) Python script} \\
			\end{array}$
		\end{figure}

		We will prefer scripts because a file can be saved, reviewed, rerun, and
		shared.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{First Script}

		Create a file named \texttt{hello.py}:
		\begin{lstlisting}
print("Hello world")
		\end{lstlisting}

		Run it from the same directory:
		\begin{lstlisting}[style=shellstyle]
python3 hello.py
		\end{lstlisting}

		The terminal prints:
		\begin{lstlisting}[style=shellstyle]
Hello world
		\end{lstlisting}
	\end{frame}

	\section{Basic I/O}
	\begin{frame}[fragile]
		\frametitle{print()}

		\texttt{print()} writes values to the terminal.
		\begin{lstlisting}
print("Hello world")
print("The answer is", 42)
		\end{lstlisting}

		The output is:
		\begin{lstlisting}[style=shellstyle]
Hello world
The answer is 42
		\end{lstlisting}

		Try with other strings and numbers.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{input()}

		\texttt{input()} waits for the user and returns a string.
		\begin{lstlisting}
name = input("Name: ")
print("Hello,", name)
		\end{lstlisting}

		Example run:
		\begin{lstlisting}[style=shellstyle]
Name: Ada
Hello, Ada
		\end{lstlisting}

		Use \texttt{int()} or \texttt{float()} when text input should become a
		number.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{Variables}

		A variable name points to a value.
		\begin{lstlisting}
count = 3
price = 2.5
message = "samples"

print(count, message)
print(count * price)
		\end{lstlisting}

		You do not declare the type first. Python reads the value and keeps track
		of the type at runtime.
	\end{frame}

	\section{Data Types}
	\begin{frame}[fragile]
		\frametitle{int Type}

		\texttt{int} means integer. You can use positive numbers, negative
		numbers, and zero.
		\begin{lstlisting}
print(123)
print(-321)
print(0)
		\end{lstlisting}

		You can also write binary, octal, and hexadecimal integers:
		\begin{lstlisting}
print(0b1010)
print(0o123)
print(0xABC)
		\end{lstlisting}
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{float Type}

		\texttt{float} means a decimal number.
		\begin{lstlisting}
print(1.23)
print(-3.21)
print(0.0)
		\end{lstlisting}

		Scientific notation is useful for very large or very small numbers:
		\begin{lstlisting}
print(1.23e45)
print(6.78E-9)
		\end{lstlisting}

		Those mean $1.23 \times 10^{45}$ and $6.78 \times 10^{-9}$.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{str Type}

		\texttt{str} means string, or text.
		\begin{lstlisting}
course = "Python"
year = "2026"

print(course)
print(course + " " + year)
print(len(course))
		\end{lstlisting}

		Use quotes around string values. Both single quotes and double quotes are
		valid; choose one style and use it consistently.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{bool Type and Comparisons}

		\texttt{bool} means \texttt{True} or \texttt{False}.
		Comparisons produce boolean values.
		\begin{lstlisting}
score = 82

print(score >= 60)
print(score == 100)
print(score != 0)
		\end{lstlisting}

		Common comparison operators are \texttt{==}, \texttt{!=}, \texttt{<},
		\texttt{<=}, \texttt{>}, and \texttt{>=}.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{Arithmetic Operators}

		You can express arithmetic directly in Python.
		\begin{lstlisting}
print(3 + 4)
print(3 - 4)
print(3 * 4)
print(3 / 4)
		\end{lstlisting}

		These are equivalent to:
		\begin{example}
			$3 + 4$ \\
			$3 - 4$ \\
			$3 \times 4$ \\
			$3 \div 4$
		\end{example}
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{Power, Remainder, and Quotient}

		\begin{lstlisting}
print(3 ** 4)   # power: 81
print(15 % 4)   # remainder: 3
print(15 / 4)   # normal division: 3.75
print(15 // 4)  # floor division: 3
		\end{lstlisting}

		Use \texttt{//} when you need an integer-like quotient. Use \texttt{\%}
		when you need the remainder after division.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{List Type}

		A list stores multiple values in one variable.
		\begin{lstlisting}
empty = []
numbers = [1, 2, 3]
mixed = [1, "two", 3.0]
nested = [1, [2, 3], 4]
		\end{lstlisting}

		Lists preserve order and can be changed after they are created.
		\begin{lstlisting}
numbers.append(4)
print(numbers)
print(len(numbers))
		\end{lstlisting}
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{Indexing}

		Python counts from zero. A list with length $n$ has indices from
		$0$ to $n-1$.
		\begin{lstlisting}
values = [5, 6, 7]

print(values[0])
print(values[1])
print(values[2])
		\end{lstlisting}

		Negative indices count from the end:
		\begin{lstlisting}
print(values[-1])
		\end{lstlisting}
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{Slicing}

		A slice selects part of a sequence.
		\begin{lstlisting}
values = [10, 20, 30, 40, 50]

print(values[1:4])   # [20, 30, 40]
print(values[:3])    # [10, 20, 30]
print(values[3:])    # [40, 50]
print(values[::2])   # [10, 30, 50]
		\end{lstlisting}

		The stop index is not included. \texttt{values[1:4]} means index 1,
		index 2, and index 3.
	\end{frame}

	\section{File I/O}
	\begin{frame}[fragile]
		\frametitle{Reading a File}

		Use \texttt{with open(...)} so Python closes the file automatically.
		\begin{lstlisting}
with open("names.txt", "r", encoding="utf-8") as handle:
    for line in handle:
        print(line.strip())
		\end{lstlisting}

		\texttt{line.strip()} removes the newline character at the end of each
		line.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{Writing a File}

		Mode \texttt{"w"} creates a file or replaces an existing file.
		\begin{lstlisting}
with open("result.csv", "w", encoding="utf-8") as handle:
    handle.write("sample,count\n")
    handle.write("A,3\n")
    handle.write("B,5\n")
		\end{lstlisting}

		Use mode \texttt{"a"} when you want to append to the end of an existing
		file.
	\end{frame}

	\begin{frame}
		\frametitle{Working Directory}

		Relative paths are interpreted from the current working directory, not
		necessarily from the directory where the script is saved.

		\medskip

		In class, keep scripts and practice files in one folder, then run:
		\begin{example}
			\texttt{python3 script.py}
		\end{example}

		from that folder. This makes file examples predictable.
	\end{frame}

	\section{if, for, and while}
	\begin{frame}[fragile]
		\frametitle{if Statements}

		Use \texttt{if}, \texttt{elif}, and \texttt{else} to choose a path.
		\begin{lstlisting}
score = int(input("Score: "))

if score >= 90:
    print("A")
elif score >= 80:
    print("B")
else:
    print("Keep practicing")
		\end{lstlisting}

		Indentation is part of Python syntax.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{for Loops}

		A \texttt{for} loop repeats once for each item in a sequence.
		\begin{lstlisting}
names = ["Ada", "Grace", "Guido"]

for name in names:
    print("Hello,", name)
		\end{lstlisting}

		Use \texttt{range()} when you need repeated numbers:
		\begin{lstlisting}
for i in range(3):
    print(i)
		\end{lstlisting}
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{while Loops}

		A \texttt{while} loop repeats while a condition is true.
		\begin{lstlisting}
count = 3

while count > 0:
    print(count)
    count = count - 1

print("Go")
		\end{lstlisting}

		Make sure something inside the loop changes the condition, otherwise the
		loop may never stop.
	\end{frame}

	\section{Functions}
	\begin{frame}[fragile]
		\frametitle{Defining Functions}

		A function packages reusable code behind a name.
		\begin{lstlisting}
def greet(name):
    print("Hello,", name)

greet("Ada")
greet("Grace")
		\end{lstlisting}

		Function names should describe the action they perform.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{return Values}

		Use \texttt{return} when the function should give a value back.
		\begin{lstlisting}
def celsius_to_fahrenheit(celsius):
    return celsius * 9 / 5 + 32

temperature = celsius_to_fahrenheit(25)
print(temperature)
		\end{lstlisting}

		Printing and returning are different. Returning lets the caller keep
		using the result.
	\end{frame}

	\section{Handling Exceptions}
	\begin{frame}[fragile]
		\frametitle{Exceptions}

		An exception means Python found a problem while running the program.
		\begin{lstlisting}
number = int("hello")
		\end{lstlisting}

		This raises \texttt{ValueError} because the string is not an integer.

		\medskip

		When you see a traceback, start by reading the last line. It usually says
		the exception type and the immediate problem.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{try and except}

		Use \texttt{try} and \texttt{except} when a failure is expected and you
		can recover from it.
		\begin{lstlisting}
text = input("Count: ")

try:
    count = int(text)
    print(count * 2)
except ValueError:
    print("Please type an integer.")
		\end{lstlisting}

		Catch specific exceptions so real bugs are still visible.
	\end{frame}

	\section{Modules}
	\begin{frame}[fragile]
		\frametitle{Importing Modules}

		A module is a file or package that provides Python code you can reuse.
		\begin{lstlisting}
import math

print(math.sqrt(9))
print(math.pi)
		\end{lstlisting}

		The Python standard library is installed with Python, so many useful
		modules require no extra installation.
	\end{frame}

	\begin{frame}[fragile]
		\frametitle{Writing Your Own Module}

		Save helper functions in one file:
		\begin{lstlisting}
# tools.py
def square(x):
    return x * x
		\end{lstlisting}

		Import them from another file in the same folder:
		\begin{lstlisting}
# main.py
from tools import square

print(square(5))
		\end{lstlisting}
	\end{frame}
